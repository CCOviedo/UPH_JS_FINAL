const input = ['average', 'exceptional', 'amazing'];
const vocales = ['a', 'e', 'i', 'o', 'u'];

const output = input.map((element) => {
  return element.split('').filter(function(el) {
    return vowels.includes(el.toLowerCase());
  }).join('');
});

console.log(output); // ["aeae", "eeioa", "aai"]

Callbacks: Cocido maker

Queremos crear un script llamado cocidoMaker. Este script tendrá que preparar:

  1. El setup de los ingredientes.
  2. Cortar cada ingrediente
  3. Cocer cada ingrediente
  4. Informar de cada acción en cada paso
  5. Informar a la finalización del proceso completo
  6. Los ingredientes vienen dados en un array. Bon Appetit!
let totalIngredientes = 0;
let ingredienteActual = 0;

const cocidoMaker = (ingredientes, callback) => {
  callback(ingredientes, cortar);
}

const ingredientesSetup = (ingredientes, callback) => {
  totalIngredientes = ingredientes.length;

  for (ingrediente of ingredientes) {
    callback(ingrediente, cocer);
  }
}

const cortar = (ingrediente, callback) => {
  console.log(`Cortando ingrediente ${ingrediente}`);
  callback(ingrediente, finalizar);
}

const cocer = (ingrediente, callback) => {
  console.log(`Cociendo ingrediente ${ingrediente}`);
  ingredienteActual++;
  console.log(`${ingredienteActual} / ${totalIngredientes}`);

  if (ingredienteActual == totalIngredientes) {
    callback();
  }
}

const finalizar = () => {
  console.log('El codido está listo. ¡A disfrutar!');
}

const ingredientes = ['Nabo', 'Puerro', 'Patatas', 'Garbanzos', 'Falda', 'Gallina', 'Chorizo'];

cocidoMaker(ingredientes, ingredientesSetup);

Prácticas

Calculadora con clases

Paso 1

Realiza una clase calculadora que tenga los siguientes métodos:

  • Introducir valor
  • Suma
  • Resta
  • División
  • MulGplicación
  • Potencia
  • Limpiar

Instancia la calculadora y realiza algunos cálculos desde la consola para comprobar que funciona correctamente. La calculadora deberá acumular los cálculos hasta que se limpie.

Paso 2

Haciendo uso de la calculadora del ejercicio anterior, realiza dos nuevas calculadoras que hereden de la calculadora base:

  • Calculadora Cutronga: Esta calculadora será invocada desde la consola, mostrará los resultados de las operaciones desde alertas de JS y pedirá los valores mediante prompts
  • Calculadora Visual: Esa calculadora al iniciarse pintará en el HTML botones con las acciones disponibles, un pequeño display con el resultado y un input para meter los valores.

Recuerda que debes hacer que la herencia tenga sentido. Para ello ambas calculadoras usarán los métodos sumar, restar, dividir... etc de su padre. NO IMPLEMENTES TODOS.

AYUDITA: Quizá solo tengas que sobrescribir el constructor, la función pedir valor y pintar valor
